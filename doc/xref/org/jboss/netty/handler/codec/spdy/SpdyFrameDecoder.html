27  /**
28   * Decodes {@link ChannelBuffer}s into SPDY Data and Control Frames.
29   */
30  public class SpdyFrameDecoder extends FrameDecoder {
31  
32      private final DecoderEmbedder<ChannelBuffer> headerBlockDecompressor =
33          new DecoderEmbedder<ChannelBuffer>(new ZlibDecoder(SPDY_DICT));
34  
35      public SpdyFrameDecoder() {
36          super();
37      }
38  
39      @Override
40      protected Object decode(
41              ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer)
42              throws Exception {
43  
44          // Must read common header to determine frame length
45          if (buffer.readableBytes() < SPDY_HEADER_SIZE) {
46              return null;
47          }
48  
49          // Get frame length from common header
50          int frameOffset  = buffer.readerIndex();
51          int lengthOffset = frameOffset + SPDY_HEADER_LENGTH_OFFSET;
52          int dataLength   = getUnsignedMedium(buffer, lengthOffset);
53          int frameLength  = SPDY_HEADER_SIZE + dataLength;
54  
55          // Wait until entire frame is readable
56          if (buffer.readableBytes() < frameLength) {
57              return null;
58          }
59  
60          // Read common header fields
61          boolean control = (buffer.getByte(frameOffset) & 0x80) != 0;
62          int flagsOffset = frameOffset + SPDY_HEADER_FLAGS_OFFSET;
63          byte flags = buffer.getByte(flagsOffset);
64  
65          if (control) {
66              // Decode control frame common header
67              int version = getUnsignedShort(buffer, frameOffset) & 0x7FFF;
68  
69              // Spdy versioning spec is broken
70              if (version != SPDY_VERSION) {
71                  buffer.skipBytes(frameLength);
72                  throw new SpdyProtocolException(
73                          "Unsupported version: " + version);
74              }
75  
76              int typeOffset = frameOffset + SPDY_HEADER_TYPE_OFFSET;
77              int type = getUnsignedShort(buffer, typeOffset);
78              buffer.skipBytes(SPDY_HEADER_SIZE);
79  
80              return decodeControlFrame(type, flags, buffer.readBytes(dataLength));
81          } else {
82              // Decode data frame common header
83              int streamID = getUnsignedInt(buffer, frameOffset);
84              buffer.skipBytes(SPDY_HEADER_SIZE);
85  
86              SpdyDataFrame spdyDataFrame = new DefaultSpdyDataFrame(streamID);
87              spdyDataFrame.setLast((flags & SPDY_DATA_FLAG_FIN) != 0);
88              spdyDataFrame.setCompressed((flags & SPDY_DATA_FLAG_COMPRESS) != 0);
89              spdyDataFrame.setData(buffer.readBytes(dataLength));
90  
91              return spdyDataFrame;
92          }
93      }

95      private Object decodeControlFrame(int type, byte flags, ChannelBuffer data)
96              throws Exception {
97          int streamID;
98          boolean last;
99  
100         switch (type) {
101         case SPDY_SYN_STREAM_FRAME:
102             if (data.readableBytes() < 12) {
103                 throw new SpdyProtocolException(
104                         "Received invalid SYN_STREAM control frame");
105             }
106             streamID = getUnsignedInt(data, data.readerIndex());
107             int associatedToStreamID = getUnsignedInt(data, data.readerIndex() + 4);
108             byte priority = (byte) (data.getByte(data.readerIndex() + 8) >> 6 & 0x03);
109             data.skipBytes(10);
110 
111             SpdySynStreamFrame spdySynStreamFrame =
112                 new DefaultSpdySynStreamFrame(streamID, associatedToStreamID, priority);
113 
114             last = (flags & SPDY_FLAG_FIN) != 0;
115             boolean unid = (flags & SPDY_FLAG_UNIDIRECTIONAL) != 0;
116             spdySynStreamFrame.setLast(last);
117             spdySynStreamFrame.setUnidirectional(unid);
118 
119             decodeHeaderBlock(spdySynStreamFrame, decompress(data));
120 
121             return spdySynStreamFrame;
122 
123         case SPDY_SYN_REPLY_FRAME:
124             if (data.readableBytes() < 8) {
125                 throw new SpdyProtocolException(
126                         "Received invalid SYN_REPLY control frame");
127             }
128             streamID = getUnsignedInt(data, data.readerIndex());
129             data.skipBytes(6);
130 
131             SpdySynReplyFrame spdySynReplyFrame =
132                 new DefaultSpdySynReplyFrame(streamID);
133 
134             last = (flags & SPDY_FLAG_FIN) != 0;
135             spdySynReplyFrame.setLast(last);
136 
137             decodeHeaderBlock(spdySynReplyFrame, decompress(data));
138 
139             return spdySynReplyFrame;
140 
141         case SPDY_RST_STREAM_FRAME:
142             if (flags != 0 || data.readableBytes() != 8) {
143                 throw new SpdyProtocolException(
144                         "Received invalid RST_STREAM control frame");
145             }
146             streamID = getUnsignedInt(data, data.readerIndex());
147             int statusCode = getSignedInt(data, data.readerIndex() + 4);
148             if (statusCode == 0) {
149                 throw new SpdyProtocolException(
150                         "Received invalid RST_STREAM status code");
151             }
152 
153             return new DefaultSpdyRstStreamFrame(streamID, statusCode);
154 
155         case SPDY_SETTINGS_FRAME:
156             if (data.readableBytes() < 4) {
157                 throw new SpdyProtocolException(
158                         "Received invalid SETTINGS control frame");
159             }
160             // Each ID/Value entry is 8 bytes
161             // The number of entries cannot exceed SPDY_MAX_LENGTH / 8;
162             int numEntries = getUnsignedInt(data, data.readerIndex());
163             if ((numEntries > (SPDY_MAX_LENGTH - 4) / 8) ||
164                 (data.readableBytes() != numEntries * 8 + 4)) {
165                 throw new SpdyProtocolException(
166                         "Received invalid SETTINGS control frame");
167             }
168             data.skipBytes(4);
169 
170             SpdySettingsFrame spdySettingsFrame = new DefaultSpdySettingsFrame();
171 
172             boolean clear = (flags & SPDY_SETTINGS_CLEAR) != 0;
173             spdySettingsFrame.setClearPreviouslyPersistedSettings(clear);
174 
175             for (int i = 0; i < numEntries; i ++) {
176                 // Chromium Issue 79156
177                 // SPDY setting ids are not written in network byte order
178                 // Read id assuming the architecture is little endian
179                 int ID = (data.readByte() & 0xFF) |
180                          (data.readByte() & 0xFF) << 8 |
181                          (data.readByte() & 0xFF) << 16;
182                 byte ID_flags = data.readByte();
183                 int value = getSignedInt(data, data.readerIndex());
184                 data.skipBytes(4);
185 
186                 if (!(spdySettingsFrame.isSet(ID))) {
187                     boolean persistVal = (ID_flags & SPDY_SETTINGS_PERSIST_VALUE) != 0;
188                     boolean persisted  = (ID_flags & SPDY_SETTINGS_PERSISTED) != 0;
189                     spdySettingsFrame.setValue(ID, value, persistVal, persisted);
190                 }
191             }
192 
193             return spdySettingsFrame;
194 
195         case SPDY_NOOP_FRAME:
196             if (data.readableBytes() != 0) {
197                 throw new SpdyProtocolException(
198                         "Received invalid NOOP control frame");
199             }
200 
201             return null;
202 
203         case SPDY_PING_FRAME:
204             if (data.readableBytes() != 4) {
205                 throw new SpdyProtocolException(
206                         "Received invalid PING control frame");
207             }
208             int ID = getSignedInt(data, data.readerIndex());
209 
210             return new DefaultSpdyPingFrame(ID);
211 
212         case SPDY_GOAWAY_FRAME:
213             if (data.readableBytes() != 4) {
214                 throw new SpdyProtocolException(
215                         "Received invalid GOAWAY control frame");
216             }
217             int lastGoodStreamID = getUnsignedInt(data, data.readerIndex());
218 
219             return new DefaultSpdyGoAwayFrame(lastGoodStreamID);
220 
221         case SPDY_HEADERS_FRAME:
222             // Protocol allows length 4 frame when there are no name/value pairs
223             if (data.readableBytes() == 4) {
224                 streamID = getUnsignedInt(data, data.readerIndex());
225                 return new DefaultSpdyHeadersFrame(streamID);
226             }
227 
228             if (data.readableBytes() < 8) {
229                 throw new SpdyProtocolException(
230                         "Received invalid HEADERS control frame");
231             }
232             streamID = getUnsignedInt(data, data.readerIndex());
233             data.skipBytes(6);
234 
235             SpdyHeadersFrame spdyHeadersFrame = new DefaultSpdyHeadersFrame(streamID);
236 
237             decodeHeaderBlock(spdyHeadersFrame, decompress(data));
238 
239             return spdyHeadersFrame;
240 
241         case SPDY_WINDOW_UPDATE_FRAME:
242             return null;
243 
244         default:
245             return null;
246         }
247     }

249     private ChannelBuffer decompress(ChannelBuffer compressed) throws Exception {
250         if ((compressed.readableBytes() == 2) &&
251             (compressed.getShort(compressed.readerIndex()) == 0)) {
252             return compressed;
253         }
254         headerBlockDecompressor.offer(compressed);
255         return headerBlockDecompressor.poll();
256     }

258     private void decodeHeaderBlock(SpdyHeaderBlock headerFrame, ChannelBuffer headerBlock)
259             throws Exception {
260         if (headerBlock.readableBytes() < 2) {
261             throw new SpdyProtocolException(
262                     "Received invalid header block");
263         }
264         int numEntries = getUnsignedShort(headerBlock, headerBlock.readerIndex());
265         headerBlock.skipBytes(2);
266         for (int i = 0; i < numEntries; i ++) {
267             if (headerBlock.readableBytes() < 2) {
268                 throw new SpdyProtocolException(
269                         "Received invalid header block");
270             }
271             int nameLength = getUnsignedShort(headerBlock, headerBlock.readerIndex());
272             headerBlock.skipBytes(2);
273             if (nameLength == 0) {
274                 headerFrame.setInvalid();
275                 return;
276             }
277             if (headerBlock.readableBytes() < nameLength) {
278                 throw new SpdyProtocolException(
279                         "Received invalid header block");
280             }
281             byte[] nameBytes = new byte[nameLength];
282             headerBlock.readBytes(nameBytes);
283             String name = new String(nameBytes, "UTF-8");
284             if (headerFrame.containsHeader(name)) {
285                 throw new SpdyProtocolException(
286                         "Received duplicate header name: " + name);
287             }
288             if (headerBlock.readableBytes() < 2) {
289                 throw new SpdyProtocolException(
290                         "Received invalid header block");
291             }
292             int valueLength = getUnsignedShort(headerBlock, headerBlock.readerIndex());
293             headerBlock.skipBytes(2);
294             if (valueLength == 0) {
295                 headerFrame.setInvalid();
296                 return;
297             }
298             if (headerBlock.readableBytes() < valueLength) {
299                 throw new SpdyProtocolException(
300                         "Received invalid header block");
301             }
302             byte[] valueBytes = new byte[valueLength];
303             headerBlock.readBytes(valueBytes);
304             int index = 0;
305             int offset = 0;
306             while (index < valueBytes.length) {
307                 while (index < valueBytes.length && valueBytes[index] != (byte) 0) {
308                     index ++;
309                 }
310                 if (index < valueBytes.length && valueBytes[index + 1] == (byte) 0) {
311                     // Received multiple, in-sequence NULL characters
312                     headerFrame.setInvalid();
313                     return;
314                 }
315                 String value = new String(valueBytes, offset, index - offset, "UTF-8");
316                 headerFrame.addHeader(name, value);
317                 index ++;
318                 offset = index;
319             }
320         }
321     }
